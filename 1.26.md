## [101. 对称二叉树](https://leetcode.cn/problems/symmetric-tree/)
### 递归法
```c++
class Solution {
public:
    bool compare(TreeNode* left, TreeNode* right) {//left和right分别指向二叉树中对称的两个节点，特殊的：根节点时，left和right都指向root
        if (left == nullptr && right == nullptr) {
            return true;
        }
        else if (left == nullptr && right != nullptr) {
            return false;
        }
        else if (left != nullptr && right == nullptr) {
            return false;
        }//比较存在null的情况
        else if (left->val != right->val) {
            return false;
        }
        else {
            //当对称节点的值相同时才去检查他们的孩子，也就是下一层是否对称
            bool outside = compare(left->left, right->right);
            bool inside = compare(left->right, right->left);
            return outside && inside;
        }
    }
    bool isSymmetric(TreeNode* root) {
        return compare(root, root);
    }
};
```
### 迭代法：使用队列：半层序遍历
`示意图`

`代码`
```c++
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        if (root == nullptr) {
            return true;
        }
        TreeNode* left = root->left;
        TreeNode* right = root->right;
        queue<TreeNode*> que;
        que.push(left);
        que.push(right);
        while (!que.empty()) {
            left = que.front();
            que.pop();
            right = que.front();
            que.pop();
            if (left == nullptr && right == nullptr) {
                continue;
            }
            else if (left == nullptr || right == nullptr || left->val != right->val) {
                return false;//任意只有一边为null，或者左右不等
            }
            else {//否则就只有都不为空，且都相等了
                que.push(left->left);
                que.push(right->right);
                que.push(left->right);
                que.push(right->left);
            }
        }
        return true;//队列为空，说明比完所有节点了。没有返回false，说明对称节点都相等，返回true
    }
};
```

```c++
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        if (root == nullptr) {
            return true;
        }
        TreeNode* left = root->left;
        TreeNode* right = root->right;
        stack<TreeNode*> st;
        st.push(left);
        st.push(right);
        while (!st.empty()) {
            right = st.top();
            st.pop();
            left = st.top();
            st.pop();
            if (left == nullptr && right == nullptr) {
                continue;
            }
            else if (left == nullptr || right == nullptr || left->val != right->val) {

    }
```

## [100. 相同的树](https://leetcode.cn/problems/same-tree/)
### 递归法
```c++
class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        if (p == nullptr && q == nullptr) {
            return true;
        }
        else if (p == nullptr || q == nullptr || p->val != q->val) {
            return false;
        }
        else {
            //只剩下都不为空，且值都相等
            bool left = isSameTree(p->left, q->left);
            bool right = isSameTree(p->right, q->right);
            return left && right;
        }
    }
};
```
### 迭代法
```c++
class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        stack<TreeNode*> st;
        st.push(p);
        st.push(q);
        while (!st.empty()) {
            TreeNode* left = st.top();
            st.pop();
            TreeNode* right = st.top();
            st.pop();
            if (left == nullptr && right == nullptr) {
                continue;
            }
            else if (left == nullptr || right == nullptr || left->val != right->val) {
                return false;
            }
            else {
                st.push(left->left);
                st.push(right->left);
                st.push(left->right);
                st.push(right->right);
            }
        }
        return true;
    }
};
```

## [572. 另一棵树的子树](https://leetcode.cn/problems/subtree-of-another-tree/)
`当root和subroot都可以为空时：`
```c++
class Solution {
public:
    bool isSame(TreeNode* root, TreeNode* subRoot) {//比较是否完全相等
        if (root == nullptr && subRoot == nullptr) {
            return true;
        }
        else if (root == nullptr || subRoot == nullptr || root->val != subRoot->val) {
            return false;
        }
        else {
            return isSame(root->left, subRoot->left) && isSame(root->right, subRoot->right);
        }
    }
    bool isSubtree(TreeNode* root, TreeNode* subRoot) {
        if (root == nullptr && subRoot!= nullptr) return false;
        if (isSame(root, subRoot)) {//如果该树和子树一样，返回true
            return true;
        }
        else if (root->left != nullptr && root->right != nullptr) {//如果该树和子树不一样，且该树的左右儿子都存在，那么比较他的左右儿子是否与子树一样
            return isSubtree(root->left, subRoot) || isSubtree(root->right, subRoot);
        }
        else if (root->left == nullptr && root->right == nullptr) {//如果该树与子书不一样，没有左右儿子，那么该树及其子树都不可能匹配
            return false;
        }
        else if (root->left == nullptr) {//如果该树与子树不一样，但是仅有左儿子，就去判断左儿子是否与子树一样
            return isSubtree(root->right, subRoot);
        }
        else {//如果该树与子树不一样，但是仅有右儿子，就去判断右儿子是否与子树一样
            return isSubtree(root->left, subRoot);
        }
    }
};
```

```c++
class Solution {

            return true;
        }
        else if (root == nullptr || subRoot == nullptr || root->val != subRoot->val) {

            return isSame(root->left, subRoot->left) && isSame(root->right, subRoot->right);
        }
    }
    bool isSubtree(TreeNode* root, TreeNode* subRoot) {
        if (root == nullptr) return false;//由于subroot>1，不用比了，肯定不匹配
        else if (isSame(root, subRoot)) {
            return true;
        }
        else {
            return isSubtree(root->left, subRoot) || isSubtree(root->right, subRoot);
        }

    }
};
```
